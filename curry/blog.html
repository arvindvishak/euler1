-- Euler1 in Curry

-- write an integer:
putInt n = if n<0 then putChar '-' >> putInt (-n) else
           if n<=9 then putChar(chr(ord '0' + n))
                   else putInt (n `div` 10) >>
                        putChar(chr(ord '0' + n `mod` 10))

euler1 :: Int -> Int
euler1 n = foldr (+) 0 [x | x <- [0..n], x `mod` 3 == 0 || x `mod` 5 == 0]

main = putInt (euler1 999) >> putChar '\n'
